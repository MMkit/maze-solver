085       /**
086        * Bring up the open file dialog and based on the file type selected hand it
087        * off to the appropriate panel. The 2 groups of files are AI scripts and
088        * maze files. AI scripts are python .py files and maze maps can be .maz or
089        * .mz2.
090        */
091       private void doOpenFile()
092       {
093          final MenuControlled[] panels =
094          {
095             codeEditorPanel, mazeEditor
096          };
097          final JFileChooser fc = new JFileChooser();
098          fc.setAcceptAllFileFilterUsed(false);
099    
100          // Create a file filter for all our file types.
101          FileFilter allFiles = new FileFilter()
102          {
103             @Override
104             public boolean accept(File f)
105             {
106                if (f.isDirectory())
107                   return true;
108                for (MenuControlled mc : panels)
109                {
110                   if (mc.isMyFileType(f))
111                      return true;
112                }
113                return false;
114             }
115    
116             @Override
117             public String getDescription()
118             {
119                String msg = "";
120                for (MenuControlled mc : panels)
121                {
122                   msg += mc.getFileTypeDescription() + ", ";
123                }
124                return msg.substring(0, msg.length() - 2);
125             }
126          };
127          fc.addChoosableFileFilter(allFiles);
128    
129          // Add filter for each type.
130          for (final MenuControlled mc : panels)
131          {
132             fc.addChoosableFileFilter(new FileFilter()
133             {
134                @Override
135                public boolean accept(File f)
136                {
137                   if (f.isDirectory())
138                      return true;
139                   else
140                      return mc.isMyFileType(f);
141                }
142    
143                @Override
144                public String getDescription()
145                {
146                   return mc.getFileTypeDescription();
147                }
148             });
149          }
150    
151          fc.setFileFilter(allFiles); // Set initially selected.
152    
153          if (fc.showOpenDialog(PrimaryFrame.this) == JFileChooser.APPROVE_OPTION)
154          {
155             File file = fc.getSelectedFile();
156             try
157             {
158                if (file.exists() && file.canRead())
159                {
160                   // Try to open the file so it throws an exception if the file can not be read.
161                   FileInputStream fileStream = null;
162                   try
163                   {
164                      fileStream = new FileInputStream(file);
165                   }
166                   finally
167                   {
168                      if (fileStream != null)
169                         fileStream.close();
170                   }
171                   for (final MenuControlled mc : panels)
172                   {
173                      if (mc.isMyFileType(file))
174                      {
175                         mainTabs.setSelectedComponent((Component) mc);
176                         mc.open(file);
177                         return;
178                      }
179                   }
180                   // If we make it this far then nobody recognized the file type.
181                   throw new RuntimeException("The type of the selected file was not recognized.");
182                }
183                else
184                {
185                   throw new SecurityException("File may not exist.");
186                }
187             }
188             catch (Exception ex)
189             {
190                ex.printStackTrace();
191                String msg = "There was an error opening the file.\n" + ex.getLocalizedMessage();
192                JOptionPane.showMessageDialog(PrimaryFrame.this,
193                                              msg,
194                                              "File Open Error",
195                                              JOptionPane.ERROR_MESSAGE);
196             }
197          }
198       } // End open file.

212       /**
213        * Get the date that this build was created on.
214        * @return Date string or <code>Unknown</code>.
215        */
216       private String getApplicationBuildDate()
217       {
218          final String unknown = "Unknown";
219          InputStream is = null;
220          try
221          {
222             Properties prop = new Properties();
223             // This properties file is created by Ant during the build process.
224             is = Main.class.getResourceAsStream("build.properties");
225             prop.load(is);
226             return prop.getProperty("date", unknown);
227          }
228          catch (Exception e1)
229          {
230             return unknown;
231          }
232          finally
233          {
234             try
235             {
236                if (is != null)
237                   is.close();
238             }
239             catch (IOException e)
240             {
241                e.printStackTrace();
242             }
243          }
244       }

246       /**
247        * Get the subversion revision this application was built from.
248        * @return The Subversion revision number or 0 if unknown.
249        */
250       private int getApplicationRevision()
251       {
252          InputStream is = null;
253          try
254          {
255             Properties prop = new Properties();
256             // This properties file is created by Ant during the build process.
257             is = Main.class.getResourceAsStream("build.properties");
258             if (is != null)
259             {
260                prop.load(is);
261                return Integer.parseInt(prop.getProperty("revision", "0"));
262             }
263          }
264          catch (NumberFormatException e)
265          {}
266          catch (IOException e1)
267          {}
268          finally
269          {
270             try
271             {
272                if (is != null)
273                   is.close();
274             }
275             catch (IOException e)
276             {
277                e.printStackTrace();
278             }
279          }
280          return 0; // On error.
281       }

311          DisplayMode dm = GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice().getDisplayMode();

313          {
314             this.setExtendedState(JFrame.MAXIMIZED_BOTH);
315          }
316          else
317          {
318             this.setSize(DEFAULT_SIZE);
319             // Center the frame on the screen.
320             this.setLocation(dm.getWidth() / 2 - DEFAULT_SIZE.width / 2, dm.getHeight() /
321                                                                          2 -
322                                                                          DEFAULT_SIZE.height /
323                                                                          2);
324          }
325    
326          this.setTitle("Micro Mouse Maze Editor and Simulator");
327          this.setIconImage(Main.getImageResource("gui/images/logo.png").getImage());
328    
329          // Main menu bar.
330          final JMenuBar menuBar = new JMenuBar();
331          this.setJMenuBar(menuBar);
332          menuBar.add(this.fileMenu);
333    
334          //New file menu.
335          JMenu newMenu = new JMenu("New");
336          fileMenu.add(newMenu);
337    
338          //New maze.
339          JMenuItem newMaze = new JMenuItem("Maze...");
340          newMaze.addActionListener(new ActionListener()
341          {
342             @Override

361          });

366          {
367             @Override
368             public void actionPerformed(ActionEvent e)

373

375    
376          // Save

378          fileSave.addActionListener(new ActionListener()
379          {
380             @Override

392    
393          // Close file.
394          JMenuItem fileClose = new JMenuItem("Close");
395          fileClose.addActionListener(new ActionListener()
396          {
397             @Override
398             public void actionPerformed(ActionEvent e)
399             {
400                Component c = mainTabs.getSelectedComponent();
401                if (c instanceof MenuControlled)
402                {
403                   MenuControlled mc = (MenuControlled) c;

405                }
406             }
407          });
408          fileMenu.add(fileClose);
409    
410          // exit

412          JMenuItem fileExit = new JMenuItem("Exit");
413          fileMenu.add(fileExit);
414          fileExit.addActionListener(new ActionListener()
415          {
416             @Override
417             public void actionPerformed(ActionEvent e)
418             {
419                exit();
420             }

422    
423          // Help menu.
424          JMenu helpMenu = new JMenu("Help");
425          menuBar.add(helpMenu);
426    
427          //Help
428          JMenuItem helpItem = new JMenuItem("Help");
429          helpMenu.add(helpItem);
430    
431          helpItem.addActionListener(new ActionListener()
432          {
433             @Override
434             public void actionPerformed(ActionEvent e)

436                HelpInfo.createAndShowGUI();
437             }
438          });
439    
440          // Look and feel menu.
441          String defaultLAF = "";
442          JMenu lookAndFeel = new JMenu("Look And Feel");
443          helpMenu.add(lookAndFeel);
444          ButtonGroup lafBG = new ButtonGroup();
445          LookAndFeelListener lafal = new LookAndFeelListener();
446          boolean nimbusFound = false;
447          for (LookAndFeelInfo laf : UIManager.getInstalledLookAndFeels())
448          {
449             JRadioButtonMenuItem jrbmi = new JRadioButtonMenuItem(laf.getName());
450             lafBG.add(jrbmi);
451             jrbmi.addActionListener(lafal);
452             jrbmi.setActionCommand(laf.getClassName());
453             lookAndFeel.add(jrbmi);
454             if (laf.getName().equals("Nimbus"))
455             {
456                defaultLAF = laf.getClassName();

461          if (!nimbusFound)

464             Enumeration<AbstractButton> ab = lafBG.getElements();
465             while (ab.hasMoreElements())

473          helpMenu.add(this.makeUpdateCheckMenuItem());

475    
476          // Separate the about dialog.
477          helpMenu.addSeparator();
478    
479          // Help->About menu item.

482          this.mazeViewer = new MazeViewerPage();

484          this.mazeEditor = new MazeEditorPage();

486          this.add(mainTabs);

491    
492          try
493          {
494             UIManager.setLookAndFeel(defaultLAF);
495          }
496          catch (Exception ex)
497          {
498             ex.printStackTrace();
499          }
500          SwingUtilities.updateComponentTreeUI(this);
501       } // End init.

535       /**
536        * Creates a menu item that checks if you have the latest revision of the
537        * application.
538        * @return Menu item.
539        */
540       private JMenuItem makeUpdateCheckMenuItem()
541       {
542          final JMenuItem item = new JMenuItem("Check for Update...");
543          item.addActionListener(new ActionListener()
544          {
545             @Override
546             public void actionPerformed(ActionEvent e)
547             {
548                Thread t = new Thread()
549                {
550                   @Override
551                   public void run()
552                   {
553                      Reader r = null;
554                      try
555                      {
556                         r = new InputStreamReader(new java.net.URI(LATEST_REVISION_URI).toURL().openConnection().getInputStream(),
557                                                   "ASCII");
558                         CharBuffer cb = CharBuffer.allocate(1024);
559                         r.read(cb);
560                         cb.flip();
561                         System.out.println(cb.toString());
562                         final int revision = Integer.parseInt(cb.toString());
563                         final int myRevision = getApplicationRevision();
564                         if (myRevision == 0)
565                         {
566                            JOptionPane.showMessageDialog(PrimaryFrame.this,
567                                                          "Unable to determine your current version.");
568                         }
569                         else if (revision == myRevision)
570                         {
571                            JOptionPane.showMessageDialog(PrimaryFrame.this,
572                                                          "You have the latest version.");
573                         }
574                         else
575                         {
576                            JOptionPane.showMessageDialog(PrimaryFrame.this,
577                                                          "There is a newer version available.\nYou have Revision: " +
578                                                                myRevision +
579                                                                "\nLatest Revision: " +
580                                                                revision);
581                         }
582                      }
583                      catch (Exception e)
584                      {
585                         e.printStackTrace();
586                         JOptionPane.showMessageDialog(PrimaryFrame.this,
587                                                       "An error occured while checking for the latest version.\n" +
588                                                             e.toString());
589                      }
590                      finally
591                      {
592                         try
593                         {
594                            if (r != null)
595                               r.close();
596                         }
597                         catch (IOException e)
598                         {
599                            e.printStackTrace();
600                         }
601                      }
602                   }
603                };
604                t.setDaemon(true);
605                t.start();
606             }
607          });
608          return item;
609       }
