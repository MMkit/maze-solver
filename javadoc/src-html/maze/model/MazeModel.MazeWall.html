069       public MazeModel(int width, int height)
070       {
071          this.width = width + width % 2;
072          this.height = height + height % 2;
073          rwalls = new BitSet(width * (height - 1));
074          cwalls = new BitSet( (width - 1) * height);
075          setWall(1, height, EAST);
076       }

078       public void setWall(int x, int y, int dir)
079       {
080          //Check for the boundary conditions
081          //First make sure that we're in the maze
082          if ( (x < 1) || (y < 1) || (x > width) || (y > height))
083             return;
084          //Second make sure that the wall isn't implied
085          if ( ( (dir == NORTH) && (y == 1)) ||
086              ( (dir == EAST) && (x == width)) ||
087              ( (dir == SOUTH) && (y == height)) ||
088              ( (dir == WEST) && (x == 1)))
089             return;
090    
091          x--;
092          y--;
093    
094          if (dir == NORTH)
095             y--;
096          if (dir == WEST)
097             x--;
098    
099          if (dir == NORTH || dir == SOUTH)
100          {
101             if (!rwalls.get(y * width + x))
102             {
103                rwalls.set(y * width + x);
104                super.notifyListeners(new MazeCell(x + 1, y + 1));
105             }
106          }
107          else
108          {
109             if (!cwalls.get(x * height + y))
110             {
111                cwalls.set(x * height + y);
112                super.notifyListeners(new MazeCell(x + 1, y + 1));
113             }
114          }
115    
116       }

118       public boolean getWall(int x, int y, int dir)
119       {
120          //Check for the boundary conditions
121          if ( ( (dir == NORTH) && (y == 1)) ||
122              ( (dir == EAST) && (x == width)) ||
123              ( (dir == SOUTH) && (y == height)) ||
124              ( (dir == WEST) && (x == 1)))
125             return true;
126    
127          if ( (x > width) || (y > height) || (x < 1) || (y < 1))
128             return true;
129          x--;
130          y--;
131    
132          if (dir == NORTH)
133             y--;
134          if (dir == WEST)
135             x--;
136    
137          if (dir == NORTH || dir == SOUTH)
138             return rwalls.get(y * width + x);
139          else
140             return cwalls.get(x * height + y);
141       }

143       public void clearWall(int x, int y, int dir)
144       {
145          //Check for the boundary conditions
146          if ( ( (dir == NORTH) && (y == 1)) ||
147              ( (dir == EAST) && (x == width)) ||
148              ( (dir == SOUTH) && (y == height)) ||
149              ( (dir == WEST) && (x == 1)))
150             return;
151    
152          if (x > width || y > height || x < 1 || y < 1)
153             return;
154          x--;
155          y--;
156    
157          if (dir == NORTH)
158             y--;
159          if (dir == WEST)
160             x--;
161    
162          if (dir == NORTH || dir == SOUTH)
163          {
164             if (rwalls.get(y * width + x))
165             {
166                rwalls.clear(y * width + x);
167                super.notifyListeners(new MazeCell(x + 1, y + 1));
168             }
169          }
170          else
171          {
172             if (cwalls.get(x * height + y))
173             {
174                cwalls.clear(x * height + y);
175                super.notifyListeners(new MazeCell(x + 1, y + 1));
176             }
177          }
178       }

180       public boolean isLegal()
181       {
182          //There are three rules that must be upheld to be a valid maze
183          //Rule 1: There is a wall next to the starting square
184          if (!getWall(1, height, EAST) || getWall(1, height, NORTH))
185             return false;
186          //Rule 2: There is one and only one way into the center of the maze
187          //Even though I've coded this to accept a variable size in the future
188          //I shall still assume here that the total size will always be even      
189          int row = height / 2, column = width / 2;
190          if (!isCenterLegal())
191             return false;
192    
193          //Offshoot of Rule 2 is that the center of the maze is an open peg
194          if (!isCenterOpen())
195             return false;
196    
197          //Rule 3: There must be at least one wall coming from every peg
198          //Except for the lone center peg
199          for (int i = 1; i <= width; i++)
200          {
201             for (int j = 1; j <= height; j++)
202             {
203                if ( (i == row && j == column) || isPegLegal(i, j))
204                   continue;
205                return false;
206             }
207          }
208          return true;
209       }

218       public boolean isPegLegal(int x, int y)
219       {
220          //This function is designed to give a lower-level view to the isLegal
221          //method by telling about a specific peg
222    
223          //The peg location is based off of the cell that it is in the NORTHWEST
224          //corner of.  Peg (1,1) is the only peg touching cell (1,1)
225          if (x < 1 || y < 1 || x >= width || y >= height)
226             return true;
227          //Every peg should have at least one wall attached to it
228          if (getWall(x, y, SOUTH) ||
229              getWall(x, y, EAST) ||
230              getWall(x + 1, y + 1, NORTH) ||
231              getWall(x + 1, y + 1, WEST))
232          {
233             //There are only three cases with walls that must be checked
234             //First is the starting square
235             if ( (x == 1) && (y == height - 1))
236             {
237                if (getWall(1, height, EAST) && !getWall(1, height, NORTH))
238                   return true;
239                else
240                   return false;
241             }
242    
243             //Second is that it is not the center peg
244             else if (x == (width / 2) && y == (height / 2))
245                return isCenterLegal();
246    
247             //Third is the pegs surrounding the center square
248             else if ( (x <= width / 2 && y <= height / 2) &&
249                      (x >= width / 2 - 1 && y >= height / 2 - 1))
250                return isCenterOpen();
251             //There are no more reasons to check the walls
252             else
253                return true;
254          }
255          else
256          {
257             //The only exception to no wall is the center peg
258             if (x == width / 2 && y == height / 2)
259                return true;
260             else
261                return false;
262          }
263       }

265       public boolean isCenterLegal()
266       {
267          int row = height / 2;
268          int column = width / 2;
269          if (getWall(column, row, SOUTH) ||
270              getWall(column, row, EAST) ||
271              getWall(column + 1, row, SOUTH) ||
272              getWall(column, row + 1, EAST))
273             return false;
274          return true;
275       }

277       public boolean isCenterOpen()
278       {
279          int walls = 0;
280          int row = height / 2;
281          int column = width / 2;
282          if (getWall(column, row, NORTH))
283             walls++;
284          if (getWall(column, row, WEST))
285             walls++;
286          if (getWall(column + 1, row, NORTH))
287             walls++;
288          if (getWall(column + 1, row, EAST))
289             walls++;
290          if (getWall(column, row + 1, WEST))
291             walls++;
292          if (getWall(column, row + 1, SOUTH))
293             walls++;
294          if (getWall(column + 1, row + 1, EAST))
295             walls++;
296          if (getWall(column + 1, row + 1, SOUTH))
297             walls++;
298    
299          if (walls != 7)
300             return false;
301          return true;
302       }

304       /**
305        * This function is basically isLegal with the added capability of returning
306        * the peg locations that are bad.<br />
307        * There are rules that must be upheld to be a valid maze.<br />
308        * Rule 1: There is a wall next to the starting square.<br />
309        * Rule 2: There is one and only one way into the center of the maze.<br />
310        * Even though I've coded this to accept a variable size in the future I
311        * shall still assume here that the total size will always be even.
312        * @return A <code>set</code> of points where each point represents an
313        *         illegal peg.
314        */
315       public Set<Point> illegalPegs()
316       {
317          TreeSet<Point> badPoints = new TreeSet<Point>(new PointCompare());
318    
319          int column = width / 2;
320          int row = height / 2;
321    
322          if (!isCenterOpen())
323          {
324             badPoints.add(new Point(column - 1, row - 1));
325             badPoints.add(new Point(column - 1, row));
326             badPoints.add(new Point(column - 1, row + 1));
327             badPoints.add(new Point(column, row - 1));
328             badPoints.add(new Point(column, row + 1));
329             badPoints.add(new Point(column + 1, row - 1));
330             badPoints.add(new Point(column + 1, row));
331             badPoints.add(new Point(column + 1, row + 1));
332          } //All of the center pegs are connected
333    
334          //Offshoot of Rule 2 is that the center of the maze is an open peg
335          if (!isCenterLegal())
336             badPoints.add(new Point(column, row));
337    
338          //Rule 3: There must be at least one wall coming from every peg
339          //Except for the lone center peg
340          for (int i = 1; i <= width; i++)
341          {
342             for (int j = 1; j <= height; j++)
343             {
344                if (!isPegLegal(i, j))
345                   badPoints.add(new Point(i, j));
346             }
347          }
348    
349          return badPoints;
350       }

370        * Get an object that can then be checked or changed.
371        * @param cell The cell containing the wall you want.
372        * @param direction The direction of the wall you want from within the given

374        * @return A <code>MazeWall</code> of the specified location.

376       public MazeWall getWall(final MazeCell cell, final Direction direction)
377       {
378          return new MazeWall()
379          {
380             @Override
381             public boolean isSet()
382             {
383                return getWall(cell.getX(), cell.getY(), direction.getIndex());
384             }
385    
386             @Override
387             public void set(boolean value)
388             {
389                if (value)
390                   setWall(cell.getX(), cell.getY(), direction.getIndex());
391                else
392                   clearWall(cell.getX(), cell.getY(), direction.getIndex());
393             }
394          };
395       }

397       public void saveMaze(String filename, String name) throws IOException
398       {
399          //This is the formatting of the .MAZ files found online
400          //This format is 4 times more inefficient than needs be
401          //but it is a format that is already present
402          if (name == null)
403          {
404             byte[] fileContents = new byte[256];
405             for (int i = 1; i <= 16; i++)
406             {
407                for (int j = 16; j > 0; j--)
408                {
409                   MazeCell cell = new MazeCell(i, j);
410                   byte cellWalls = 0;
411                   if (this.getWall(cell, Direction.North).isSet())
412                   {
413                      cellWalls = 1;
414                   }
415                   if (this.getWall(cell, Direction.East).isSet())
416                   {
417                      cellWalls += 2;
418                   }
419                   if (this.getWall(cell, Direction.South).isSet())
420                   {
421                      cellWalls += 4;
422                   }
423                   if (this.getWall(cell, Direction.West).isSet())
424                   {
425                      cellWalls += 8;
426                   }
427    
428                   fileContents[16 * (i - 1) + (16 - j)] = cellWalls;
429                }
430             }
431    
432             //Now for the actual file i/o
433             if (filename.toLowerCase().endsWith(".maz") == false)
434                filename = filename + ".maz";
435    
436             FileOutputStream out = new FileOutputStream(filename);
437             out.write(fileContents);
438             out.close();
439          } // if (name == null)
440          else
441          {
442             if (filename.toLowerCase().endsWith(".mz2") == false)
443                filename = filename + ".mz2";
444    
445             FileOutputStream out = new FileOutputStream(filename);
446             DataOutputStream dis = new DataOutputStream(out);
447             dis.writeUTF(name);
448             dis.writeInt(width);
449             dis.writeInt(height);
450    
451             int rwallSize = rwalls.size(), cwallSize = cwalls.size();
452             int rwallArraySize = rwalls.size() / 8;
453             if (rwalls.size() % 8 > 0)
454                rwallArraySize++;
455             int cwallArraySize = cwalls.size() / 8;
456             if (cwalls.size() % 8 > 0)
457                cwallArraySize++;
458             byte[] rwallArray = new byte[rwallArraySize];
459             byte[] cwallArray = new byte[cwallArraySize];
460             for (int i = 0; i < rwallArray.length; i++)
461                rwallArray[i] = 0;
462             for (int i = 0; i < cwallArray.length; i++)
463                cwallArray[i] = 0;
464    
465             int index = 0;
466    
467             for (int i = 0; i < rwallSize; i++)
468             {
469                rwallArray[index] |= ( (rwalls.get(i) ? 1 : 0) << (i % 8));
470                if ( (i + 1) % 8 == 0)
471                   index++;
472             }
473    
474             index = 0;
475    
476             for (int i = 0; i < cwallSize; i++)
477             {
478                cwallArray[index] |= ( (cwalls.get(i) ? 1 : 0) << (i % 8));
479                if ( (i + 1) % 8 == 0)
480                   index++;
481             }
482    
483             dis.write(rwallArray);
484             dis.write(cwallArray);
485             dis.close();
486          }
487    
488       }

490       public String loadMaze(InputStream in, boolean extended) throws IOException
491       {
492          if (extended)
493          {
494             DataInputStream dis = new DataInputStream(in);
495             String name = dis.readUTF();
496             setSize(new Dimension(dis.readInt(), dis.readInt()));
497    
498             int rwallSize = rwalls.size(), cwallSize = cwalls.size();
499             for (int i = 0; i < rwallSize;)
500             {
501                byte bits = dis.readByte();
502                for (int j = 0; j < 8 && i < rwallSize; j++, i++)
503                {
504                   rwalls.set(i, (bits & 1) == 1);
505                   bits >>= 1;
506                }
507             }
508             for (int i = 0; i < cwallSize;)
509             {
510                byte bits = dis.readByte();
511                for (int j = 0; j < 8 && i < cwallSize; j++, i++)
512                {
513                   cwalls.set(i, (bits & 1) == 1);
514                   bits >>= 1;
515                }
516             }
517             return name;
518          }
519          else
520          {
521             //This is the formatting of the .MAZ files found online
522             //This format is 4 times more inefficient than needs be
523             //but it is a format that is already present
524             byte[] fileContents = new byte[256];
525             setSize(new Dimension(DEFAULT_SIZE, DEFAULT_SIZE));
526    
527             if (in.available() != 256)
528                throw new IOException("Invalid format");
529    
530             in.read(fileContents);
531    
532             for (int i = 1; i <= 16; i++)
533             {
534                for (int j = 16; j > 0; j--)
535                {
536                   byte cellWalls = fileContents[16 * (i - 1) + (16 - j)];
537                   if ( (cellWalls & 1) == 1)
538                      this.setWall(i, j, NORTH);
539                   else
540                      this.clearWall(i, j, NORTH);
541    
542                   if ( (cellWalls & 2) == 2)
543                      this.setWall(i, j, EAST);
544                   else
545                      this.clearWall(i, j, EAST);
546    
547                   if ( (cellWalls & 4) == 4)
548                      this.setWall(i, j, SOUTH);
549                   else
550                      this.clearWall(i, j, SOUTH);
551    
552                   if ( (cellWalls & 8) == 8)
553                      this.setWall(i, j, WEST);
554                   else
555                      this.clearWall(i, j, WEST);
556                } // for(int j=16; j>0; j--)
557             } // for(int i=1; i<=16; i++)
558             return null;
559          }
560       }

607       /**
608        * Used to store points in a Set.
609        */
610       private static class PointCompare implements Comparator<Point>
611       {
612          @Override
613          public int compare(Point o1, Point o2)
614          {
615             if (o1.x == o2.x)
616                if (o1.y == o2.y)
617                   return 0;
618                else
619                   return o1.y - o2.y;
620             else
621                return o1.x - o2.x;
622          }
623       }
